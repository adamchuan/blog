    (b)尽量减少性能损耗
    (c)重复进入计时累加
    {% highlight html %} <div class='mod-spm'data-spmid='123'><        <div class='child_a'></div>        <div class='child_b'></div>        <div class='child_c'></div>        <div class='child_d'></div>     </div>     <div class='mod-spm'data-spmid='456'>        <div class='child_a'></div>        <div class='child_b'></div>        <div class='child_c'></div>        <div class='child_d'></div>     </div>     <div class='mod-spm'data-spmid='789'>        <div class='child_a'></div>        <div class='child_b'></div>        <div class='child_c'></div>        <div class='child_d'></div>     </div> {% endhighlight %}

    答案：这题考点没搞懂，考的是作用域？ 每个监听事件都要有自己的一个countTime的副本，然后另外能优化性能的地方，我觉得一是清除startTime的内存，还有modCollections的length放在变量中

    {% highlight javascript %} var modCollections = document.querySelectorAll(".mod-spm"); for(var i = 0 , len = modCollections.length ; i < len ; i++){ addTimer(modCollections[i]); } function addTimer(elm){ var startTime; var countTime = 0; elm.addEventListener("mouseenter",function(e){ startTime = new Date(); }); elm.addEventListener("mouseout",function(e){ countTime += (new Date() - startTime); console.log("停留时间 : " + countTime); startTime = null; }); } {% endhighlight %}

  7. 编写一个JavaScript函数，输入指定类型的选择器(仅需支持id，class，tagName三种简单CSS选择器，无需兼容组合选择器)可以返回匹配的DOM节点，需考虑浏览器兼容性和性能。
    {% highlight javascript %} /*** @param selector {String} 传入的CSS选择器。* @return{Array}*/ var query = function(selector){ //返回查找到的节点数组return []; } {% endhighlight %}

    答案：最好是用正则写。

    {% highlight javascript %} var query = function(selector){ if(typeof document.querySelectorAll === 'function'){ return document.querySelectorAll(selector); } if(typeof selector !== 'string'){ return ; } if(selector.length === 0){ return ; } var type,newSel; if(selector[0] === '#'){ type = "id"; newSel = selector.substr(1,selector.length - 1); } else if(selector[0] === '.'){ type = 'class'; newSel = selector.substr(1,selector.length - 1); }else if(/{a-zA-Z}*/.test(selector)){ type = 'tag'; newSel = selector; } var result = []; var handler = { 'class' : function(cn){ addToResult(document.body); function addToResult(elm){ if(elm.className === cn){ result.push(elm); } var children = elm.children; var len = elm.children.length; if(len > 0){ for(var i = 0 ; i < len ;i++){ addToResult(children[i]); } } } }, 'id' : function(Id){ result.push(document.getElementById(Id)); }, 'tag' : function(tag){ result = document.getElementsByTagName(tag); } } handler[type](newSel); return result; } {% endhighlight %}
